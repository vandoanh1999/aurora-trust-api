Integration

Three Lines of Code

Start logging AI decisions and issuing credentials immediately.

# 1. Issue a credential for your AI model deployment
import requests

headers = {"Authorization": "Bearer at_your_api_key"}

# Issue Verifiable Credential
vc = requests.post("https://auroratrust.dev/v1/credentials/issue",
    json={
        "subject_id": "model-gpt4-fraud-detector-v2",
        "credential_type": "AIModelCompliance",
        "claims": {
            "risk_level": "high",
            "validated_by": "compliance-team",
            "eu_ai_act_category": "Annex III — Credit Scoring"
        },
        "expires_in_days": 365
    }, headers=headers).json()

# 2. Log every AI decision
requests.post("https://auroratrust.dev/v1/audit/log",
    json={
        "agent_id": "model-gpt4-fraud-detector-v2",
        "action": "loan_application_decision",
        "outcome": "success",
        "metadata": {"applicant_id": "anon-8823", "decision": "approved"}
    }, headers=headers)

# 3. Get compliance summary for audit report
summary = requests.get(
    "https://auroratrust.dev/v1/audit/summary/model-gpt4-fraud-detector-v2",
    headers=headers).json()
print(summary["compliance_status"])  # → "compliant"
